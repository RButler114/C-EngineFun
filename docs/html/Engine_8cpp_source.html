114void Engine::CalculateDeltaTime() {
115 auto currentTime = std::chrono::high_resolution_clock::now();
116 auto duration = std::chrono::duration_cast<std::chrono::microseconds>(currentTime - m_lastFrameTime);
117 m_deltaTime = duration.count() / 1000000.0f; // Convert to seconds
118
119 // Calculate FPS

121 m_fps = 1.0f / m_deltaTime;

127void Engine::CapFrameRate() {
128 auto frameEndTime = std::chrono::high_resolution_clock::now();
129 auto frameDuration = std::chrono::duration_cast<std::chrono::microseconds>(frameEndTime - m_frameStartTime);
130
131 auto targetFrameTime = std::chrono::microseconds(1000000 / m_targetFPS);
132
133 if (frameDuration < targetFrameTime) {
134 auto sleepTime = targetFrameTime - frameDuration;
