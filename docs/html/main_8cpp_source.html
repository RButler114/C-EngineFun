113 m_rectX -= speed * deltaTime;

116 m_rectX += speed * deltaTime;

118 if (input->IsKeyPressed(SDL_SCANCODE_UP) || input->IsKeyPressed(SDL_SCANCODE_W)) {
119 m_rectY -= speed * deltaTime;
120 }
121 if (input->IsKeyPressed(SDL_SCANCODE_DOWN) || input->IsKeyPressed(SDL_SCANCODE_S)) {
122 m_rectY += speed * deltaTime;
123 }
124
125 // Keep within bounds
126 if (m_rectX < 0) m_rectX = 0;
127 if (m_rectY < 0) m_rectY = 0;
128 if (m_rectX > 750) m_rectX = 750;
129 if (m_rectY > 550) m_rectY = 550;
130
131 // Update window title with FPS
132 static float titleUpdateTimer = 0.0f;

138 titleUpdateTimer = 0.0f;
