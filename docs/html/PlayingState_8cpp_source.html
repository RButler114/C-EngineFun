157void PlayingState::HandleInput() {
158 auto* input = GetInputManager();
159 if (!input) return;
160
161 // Pause game or return to menu
162 if (input->IsKeyJustPressed(SDL_SCANCODE_P)) {
163 std::cout << "Game paused (not implemented yet)" << std::endl;
164 }
165 if (input->IsKeyJustPressed(SDL_SCANCODE_ESCAPE)) {
166 std::cout << "Returning to menu..." << std::endl;
167 if (GetStateManager()) {
168 GetStateManager()->ChangeState(GameStateType::MENU);
169 }
170 }
171
172 // Player movement (using simple variables to bypass ECS corruption)
173 m_playerVelX = 0;
174 m_playerVelY = 0;
175
176 const float speed = 250.0f;
177
178 // Horizontal movement
179 if (input->IsKeyPressed(SDL_SCANCODE_LEFT) || input->IsKeyPressed(SDL_SCANCODE_A)) {
180 m_playerVelX = -speed;
181 }
182 if (input->IsKeyPressed(SDL_SCANCODE_RIGHT) || input->IsKeyPressed(SDL_SCANCODE_D)) {
183 m_playerVelX = speed;
184 }
185
186 // Vertical movement (constrained to ground level area)
187 if (input->IsKeyPressed(SDL_SCANCODE_UP) || input->IsKeyPressed(SDL_SCANCODE_W)) {
188 if (m_playerY > 350.0f) { // Don't go too high
189 m_playerVelY = -speed;
190 }
191 }
192 if (input->IsKeyPressed(SDL_SCANCODE_DOWN) || input->IsKeyPressed(SDL_SCANCODE_S)) {
193 if (m_playerY < 450.0f) { // Don't go below ground
194 m_playerVelY = speed;
195 }
196 }
197
198 // Keep all entities (including enemies) within reasonable bounds to prevent rendering issues
199 auto allEntities = m_entityManager->GetEntitiesWith<TransformComponent, VelocityComponent>();
200 for (Entity entity : allEntities) {
201 if (entity == m_player) continue; // Skip player, already handled above
202
203 auto* transform = m_entityManager->GetComponent<TransformComponent>(entity);
204 auto* velocity = m_entityManager->GetComponent<VelocityComponent>(entity);
205
206 if (transform && velocity) {
207 // Keep enemies within vertical bounds
208 if (transform->y < 350.0f) {
209 transform->y = 350.0f; // Sky limit
210 velocity->vy = abs(velocity->vy); // Bounce down
211 }
212 if (transform->y > 450.0f) {
213 transform->y = 450.0f; // Ground limit
214 velocity->vy = -abs(velocity->vy); // Bounce up
215 }
216
217 // Remove enemies that have moved too far left (off screen)
218 if (transform->x < m_cameraX - 200.0f) {
219 // Reset enemy position to the right side for continuous gameplay
220 transform->x = m_cameraX + 1000.0f + (rand() % 500);
221 transform->y = 400.0f + (rand() % 60) - 30.0f; // Random height variation
222 }
223 }
224 }
