284 if (!IsEntityValid(entity)) {

288 ComponentTypeID typeID = GetComponentTypeID<T>();
289 auto componentMapIt = m_components.find(typeID);

291 return nullptr;

293
294 auto componentIt = componentMapIt->second.find(entity.GetID());
295 if (componentIt == componentMapIt->second.end()) {
296 return nullptr;

299 return static_cast<T*>(componentIt->second.get());

348 auto it = m_systemMap.find(std::type_index(typeid(T)));
349 if (it != m_systemMap.end()) {
350 System* systemPtr = it->second;
351 m_systemMap.erase(it);
352
353 auto systemIt = std::find_if(m_systems.begin(), m_systems.end(),
354 [systemPtr](const std::unique_ptr<System>& sys) {
355 return sys.get() == systemPtr;
356 });
357
358 if (systemIt != m_systems.end()) {
359 m_systems.erase(systemIt);
